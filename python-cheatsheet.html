Decorators

Function Decorators

Wrap functions to extend behavior.

import time from functools import wraps def timer(func): @wraps(func) def wrapper(*args, **kwargs): start = time.time() result = func(*args, **kwargs) print(f"{func.__name__}: {time.time()-start:.2f}s") return result return wrapper @timer def slow_function(): time.sleep(1)

Exceptions

Error Handling

Handle exceptions gracefully.

try: result = risky_operation() except ValueError as e: print(f"Value error: {e}") except (TypeError, KeyError) as e: print(f"Type error: {e}") except Exception as e: logger.error(f"Unexpected: {e}") raise else: print("Success") finally: cleanup()

Classes & OOP

Class

Class Basics

Define classes with methods.

class Person: species = "Human" # Class variable def __init__(self, name, age): self.name = name # Instance var self.age = age def greet(self): return f"Hi, I'm {self.name}" def __str__(self): return f"Person({self.name})" p = Person("Alice", 30) print(p.greet())
Inheritance

Inheritance

Create specialized classes.

class Employee(Person): def __init__(self, name, age, emp_id): super().__init__(name, age) self.emp_id = emp_id def greet(self): base = super().greet() return f"{base}, ID: {self.emp_id}" # Multiple inheritance class A: pass class B: pass class C(A, B): pass
Properties

Properties & Methods

Controlled attribute access.

class Circle: def __init__(self, radius): self._radius = radius @property def radius(self): return self._radius @radius.setter def radius(self, value): if value < 0: raise ValueError("Negative!") self._radius = value @property def area(self): return 3.14159 * self._radius ** 2

Concurrency

Threading

Threading

Run tasks concurrently (I/O bound).

import threading from concurrent.futures import ThreadPoolExecutor def task(n): return n * 2 # Thread pool with ThreadPoolExecutor(max_workers=4) as executor: futures = [executor.submit(task, i) for i in range(10)] results = [f.result() for f in futures] # Thread t = threading.Thread(target=task, args=(5,)) t.start() t.join()
Multiprocessing

Multiprocessing

Parallel execution (CPU bound).

from multiprocessing import Pool from concurrent.futures import ProcessPoolExecutor def heavy_task(n): return sum(i**2 for i in range(n)) # Process pool with Pool(4) as pool: results = pool.map(heavy_task, [1000]*10) # ProcessPoolExecutor with ProcessPoolExecutor(max_workers=4) as executor: results = list(executor.map(heavy_task, data))
Async

Async/Await

Asynchronous programming.

import asyncio import aiohttp async def fetch(url): async with aiohttp.ClientSession() as session: async with session.get(url) as resp: return await resp.text() async def main(): urls = ['url1', 'url2', 'url3'] tasks = [fetch(url) for url in urls] results = await asyncio.gather(*tasks) return results asyncio.run(main())
Queue

Queue

Thread-safe data sharing.

from queue import Queue from threading import Thread def producer(q): for i in range(5): q.put(i) def consumer(q): while True: item = q.get() print(f"Processing {item}") q.task_done() q = Queue() Thread(target=producer, args=(q,)).start() Thread(target=consumer, args=(q,)).start() q.join()
