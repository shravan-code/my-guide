df = spark.read.csv("path/to/file.csv")

df = spark.read.csv("path/to/file.csv", header=True)

schema = StructType([
    StructField("id", IntegerType(), False),
    StructField("name", StringType(), True),
    StructField("salary", IntegerType(), True),
    StructField("department", StringType(), True)
])

df = spark.read.csv("path/to/file.csv", header=True, schema=schema)

# Various options
df = spark.read.csv(
    "path/to/file.csv",
    header=True,
    schema=schema,
    sep=",",                    # Separator
    encoding="UTF-8",          # File encoding
    quote='"',                  # Quote character
    escape='\\',               # Escape character
    comment=None,              # Lines starting with this will be skipped
    nullValue="NA",            # Represent null values
    emptyValue="",             # Represent empty values
    nanValue="NaN",            # Represent NaN values
    dateFormat="yyyy-MM-dd",   # Date format
    timestampFormat="yyyy-MM-dd'T'HH:mm:ss",  # Timestamp format
    maxColumns=1000,           # Max columns
    maxCharsPerColumn=10000,   # Max chars per column
    maxMalformedLogPartitions=10  # Partitions for error handling
)

# Multi-line CSV
df = spark.read.csv("path/to/multiline.csv", header=True, multiLine=True)

# Handling bad records
df = spark.read.csv(
    "path/to/file.csv",
    header=True,
    mode="DROPMALFORMED"  # PERMISSIVE, DROPMALFORMED, FAILFAST
)

⚠️ CSV Common Issues & Solutions

▼ 
# ❌ ISSUE 1: Slow schema inference
df = spark.read.csv("large_file.csv", header=True, inferSchema=True)
# Scans data twice - once for schema, once for data

# ❌ ISSUE 2: Malformed records causing failures
df = spark.read.csv("messy_data.csv", header=True, mode="FAILFAST")
# Fails on first bad record

# ❌ ISSUE 3: Special characters breaking parsing
df = spark.read.csv("data.csv", header=True)
# Fails if text contains separator (e.g., "Hello, World")
# ✅ SOLUTION 1: Define schema explicitly
schema = StructType([
    StructField("id", IntegerType(), True),
    StructField("name", StringType(), True),
    StructField("salary", DoubleType(), True),
    StructField("department", StringType(), True),
    StructField("hire_date", DateType(), True)
])
df = spark.read.csv("large_file.csv", header=True, schema=schema)

# ✅ SOLUTION 2: Handle bad records gracefully
df = spark.read.csv("messy_data.csv", header=True, mode="DROPMALFORMED")
# OR: Parse with permissive mode
df = spark.read.csv("messy_data.csv", header=True, mode="PERMISSIVE")

# ✅ SOLUTION 3: Configure quote/escape handling
df = spark.read.csv(
    "data.csv",
    header=True,
    quote='"',
    escape='\\',
    sep=","
)

# Check for bad records
print(f"Total rows: {df.count()}")
print(f"Rows with nulls: {df.filter(df['id'].isNull()).count()}")

📦 Parquet File Handling

✨ Why Parquet?

• Columnar format - read only needed columns

• Built-in compression (Snappy, Gzip)

• Efficient encoding and predicate pushdown

• Schema evolution support

• Up to 10x faster than CSV for analytical queries

🔍 Reading & Writing Parquet

▼ 
# ============== READING PARQUET ==============

# Basic reading
df = spark.read.parquet("path/to/data.parquet")

# Read multiple files
df = spark.read.parquet("s3://bucket/prefix/*.parquet")
df = spark.read.parquet("path/to/data/")

# Merge schema from multiple files
df = spark.read.option("mergeSchema", "true").parquet("path/to/data/")

# Read specific columns (predicate pushdown)
df = spark.read.parquet("path/to/data.parquet").select("id", "name", "salary")

# Read with filters (partition pruning)
df = spark.read.parquet("path/to/data.parquet").filter("year = 2024")

# ============== WRITING PARQUET ==============

# Basic writing
df.write.parquet("path/to/output.parquet")

# With compression
df.write.mode("overwrite").parquet(
    "path/to/output.parquet",
    compression="snappy"  # none, snappy, gzip, lzo, zstd
)

# With partitioning (creates subdirectories)
df.write.mode("overwrite") \
    .partitionBy("year", "month", "department") \
    .parquet("path/to/output/")

# With specific row group size
df.write.option("parquet.block.size", "134217728").parquet("path/")  # 128MB

# Write with schema merge
df.write.mode("overwrite").option("mergeSchema", "true").parquet("path/")

⚠️ Parquet Common Issues & Solutions

▼ 
# ❌ ISSUE 1: Too many small files
df.write.parquet("path/to/output")
# Creates thousands of small files

# ❌ ISSUE 2: Partition column type issues
df.write.partitionBy("year", "month").parquet("path/")
# year=2024.0, month=1 instead of year=2024, month=1

# ❌ ISSUE 3: Schema mismatch
df1 = spark.read.parquet("path/to/data")
df2 = spark.read.parquet("path/to/data")
# Columns have different types but no error until query
# ✅ SOLUTION 1: Coalesce before write
df.repartition(10).write.parquet("path/to/output")
# Creates exactly 10 files

# ✅ SOLUTION 2: Cast partition columns to integers
df = df.withColumn("year", col("year").cast("int"))
df = df.withColumn("month", col("month").cast("int"))
df.write.partitionBy("year", "month").parquet("path/")

# ✅ SOLUTION 3: Merge schemas and validate
df = spark.read.option("mergeSchema", "true").parquet("path/to/data")

# Validate schema compatibility
expected_schema = StructType([...])
if df.schema != expected_schema:
    raise ValueError("Schema mismatch!")

# Check for schema evolution
print("Schema:", df.schema.simpleString())

⚠️ JSON Common Issues & Solutions

▼ 
# ❌ ISSUE 1: Multi-line JSON not parsed
df = spark.read.json("multiline_data.json")
# Empty DataFrame - each object spans multiple lines

# ❌ ISSUE 2: Mixed schemas in JSON lines
# Line 1: {"name": "Alice", "age": 30}
# Line 2: {"name": "Bob", "salary": 5000}
# Result: all nulls due to schema conflict

# ❌ ISSUE 3: Large JSON objects causing memory issues
df = spark.read.json("huge_data.json")
# Entire JSON parsed into memory
# ✅ SOLUTION 1: Enable multiLine option
df = spark.read.option("multiLine", "true").json("multiline_data.json")

# ✅ SOLUTION 2: Handle mixed schemas with permissive mode
df = spark.read.option("mode", "PERMISSIVE").json("mixed_data.json")
# Or drop records with schema conflicts
df = spark.read.option("columnNameOfCorruptRecord", "_corrupt").json("mixed_data.json")

# Filter valid records
valid_df = df.filter(col("_corrupt").isNull())

# ✅ SOLUTION 3: Process in batches using RDD
def process_json_batch(json_string):
    import json
    records = json.loads(json_string)
    return process(records)

rdd = spark.sparkContext.textFile("huge_data.json")
processed = rdd.map(process_json_batch)

⚠️ JDBC Common Issues & Solutions

▼ 
# ❌ ISSUE 1: Loading entire table into single partition
df = spark.read.jdbc(url, "large_table", properties=props)
# All data in one partition - slow

# ❌ ISSUE 2: Connection timeout
df = spark.read.jdbc(url, "table", properties=props)
# Fails after default timeout

# ❌ ISSUE 3: Wrong data type mapping
# Date columns become strings, decimals lose precision
# ✅ SOLUTION 1: Parallel read with partition column
df = spark.read.jdbc(
    url, "large_table",
    column="id",
    lowerBound=1,
    upperBound=10000000,
    numPartitions=20,  # 20 parallel reads
    properties=props
)

# ✅ SOLUTION 2: Increase timeout and fetch size
df = spark.read \
    .option("fetchsize", "10000") \
    .option("queryTimeout", 300) \
    .jdbc(url, "table", properties=props)

# ✅ SOLUTION 3: Explicit type mapping
df = spark.read \
    .option("dateFormat", "yyyy-MM-dd") \
    .option("timestampFormat", "yyyy-MM-dd HH:mm:ss") \
    .option("decimalDigits", 10) \
    .jdbc(url, "table", properties=props)

# Convert types after reading
df = df.withColumn("date_col", col("date_col").cast("date"))
df = df.withColumn("decimal_col", col("decimal_col").cast("decimal(10,2)"))

🏔️ Delta Lake Handling

# ============== DELTA LAKE ==============
# Requires: delta-core package
# spark-submit --packages io.delta:delta-core_2.12:2.4.0

from delta.tables import DeltaTable

# Reading Delta table
df = spark.read.format("delta").load("path/to/delta-table")

# Time travel - read specific version
df_v1 = spark.read.format("delta").option("versionAsOf", 1).load("path/to/delta")
df_v2 = spark.read.format("delta").option("timestampAsOf", "2024-01-15").load("path/to/delta")

# Writing to Delta
df.write.format("delta").mode("overwrite").save("path/to/delta-table")

# Merge (Upsert)
delta_table = DeltaTable.forPath(spark, "path/to/delta-table")
updates_df = spark.createDataFrame([...])  # New data

delta_table.upsert(updates_df, "target.id = source.id")

# Full merge with conditions
delta_table.upsert(updates_df) \
    .whenMatchedUpdate(
        condition="target.salary != source.salary",
        set={"name": "source.name", "salary": "source.salary"}
    ) \
    .whenMatchedDelete(condition="source.status = 'inactive'") \
    .whenNotMatchedInsertAll() \
    .execute()

# Delete
delta_table.delete(col("status") == "inactive")

# Update
delta_table.update(
    condition=col("salary") > 100000,
    set={"salary": col("salary") * 1.1}
)

# Vacuum old versions
delta_table.vacuum(retentionHours=168)  # Delete versions older than 7 days

🏆 Ranking Functions

from pyspark.sql.window import Window
from pyspark.sql.functions import row_number, rank, dense_rank, percent_rank, ntile

# ============== RANKING FUNCTIONS ==============

# Define window specification
window_spec = Window.partitionBy("department").orderBy(col("salary").desc())

# Row Number - Sequential numbering (no gaps)
employees.withColumn("row_number", row_number().over(window_spec)).show()

# Output:
# +----+--------+--------+------+----------+
# |  id|    name|department|salary|row_number|
# +----+--------+--------+------+----------+
# |   4|   Diana|Engineering|  8000|         1|
# |  10|    Jack|Engineering|  7800|         2|
# |   2|     Bob|Engineering|  7500|         3|
# |   7|   Grace|Engineering|  7200|         4|
# +----+--------+--------+------+----------+

# Rank - With gaps (1, 2, 2, 4)
employees.withColumn("rank", rank().over(window_spec)).show()

# Dense Rank - Without gaps (1, 2, 2, 3)
employees.withColumn("dense_rank", dense_rank().over(window_spec)).show()

# Percent Rank - Percentage position (0.0 to 1.0)
employees.withColumn("percent_rank", percent_rank().over(window_spec)).show()

# NTile - Divide into buckets (e.g., ntile(4) for quartiles)
employees.withColumn("quartile", ntile(4).over(window_spec)).show()

# ============== TOP N PER GROUP ==============
# Get top 2 earners per department
from pyspark.sql.functions import col

window_top2 = Window.partitionBy("department").orderBy(col("salary").desc())
top_earners = employees.withColumn("rn", row_number().over(window_top2)) \
    .filter(col("rn") <= 2) \
    .drop("rn")

# ============== BOTTOM N PER GROUP ==============
window_bottom2 = Window.partitionBy("department").orderBy(col("salary").asc())
bottom_earners = employees.withColumn("rn", row_number().over(window_bottom2)) \
    .filter(col("rn") <= 2) \
    .drop("rn")

📊 Aggregate Functions in Windows

from pyspark.sql.functions import sum, avg, min, max, count, collect_list

# ============== AGGREGATE FUNCTIONS ==============

# Window with all rows in partition
window_all = Window.partitionBy("department")

# Running sum within department
employees.withColumn("dept_total", sum("salary").over(window_all)).show()

# Running average
employees.withColumn("dept_avg", avg("salary").over(window_all)).show()

# Running min/max
employees.withColumn("dept_min", min("salary").over(window_all)) \
         .withColumn("dept_max", max("salary").over(window_all)).show()

# Count within partition
employees.withColumn("dept_count", count("*").over(window_all)).show()

# Collect list
employees.withColumn("names", collect_list("name").over(window_all)).show()

# ============== RUNNING TOTALS (Cumulative) ==============

# Define window with ordering for running calculations
window_cumulative = Window.partitionBy("department") \
    .orderBy("hire_date") \
    .rowsBetween(Window.unboundedPreceding, Window.currentRow)

employees.withColumn("hire_date", col("hire_date").cast("date")) \
    .withColumn("running_salary_sum", sum("salary").over(window_cumulative)) \
    .withColumn("running_avg", avg("salary").over(window_cumulative)) \
    .withColumn("running_count", count("*").over(window_cumulative)).show()

# ============== ROW FRAME SPECIFICATIONS ==============
# rowsBetween(start, end) where:
# Window.currentRow = 0
# Window.unboundedPreceding = -inf
# Window.unboundedFollowing = +inf

# Current row only

df.foreach(lambda row: process(row))  # Runs on executors

# OR use rdd.map for parallel processing
df.rdd.map(lambda row: process(row)).collect()

# 4. Avoid cross joins, use partitioned joins
# Instead of: df1.crossJoin(df2)
# Use: df1.join(df2, "key").repartition(n)

# 5. Reduce network traffic with select
result = df.join(other, "key").select("id", "name")  # Only needed columns
