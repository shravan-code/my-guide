Python Methods

Built-in functions, string methods, and list operations.

Built-in Methods Basic

Python provides numerous built-in methods for common data types like strings, lists, dictionaries, and more.

String Methods

s = "Hello, World!" # Case methods s.upper() # "HELLO, WORLD!" s.lower() # "hello, world!" s.title() # "Hello, World!" s.capitalize() # "Hello, world!" # Search methods s.find("World") # 7 s.count("l") # 3 s.startswith("Hello") # True s.endswith("!") # True # Transform methods s.strip() # Remove whitespace s.replace("World", "Python") s.split(", ") # ['Hello', 'World!']

List Methods

lst = [1, 2, 3] # Adding elements lst.append(4) # [1, 2, 3, 4] lst.extend([5, 6]) # [1, 2, 3, 4, 5, 6] lst.insert(0, 0) # [0, 1, 2, 3, 4, 5, 6] # Removing elements lst.remove(3) # Remove by value lst.pop() # Remove and return last lst.pop(0) # Remove by index lst.clear() # [] # Other methods lst.sort() # Sort in place lst.reverse() # Reverse in place lst.index(2) # Find index of value lst.count(1) # Count occurrences

Dictionary Methods

d = {'a': 1, 'b': 2} # Access methods d.get('a') # 1 d.get('c', 0) # 0 (default) # Modification d.update({'c': 3}) d.setdefault('d', 4) # View methods d.keys() # dict_keys(['a', 'b', 'c', 'd']) d.values() # dict_values([1, 2, 3, 4]) d.items() # dict_items([('a', 1), ...]) # Remove methods d.pop('a') # Remove and return value d.popitem() # Remove last item d.clear() # Empty dict

Set Methods

s1 = {1, 2, 3} s2 = {3, 4, 5} # Set operations s1.union(s2) # {1, 2, 3, 4, 5} s1.intersection(s2) # {3} s1.difference(s2) # {1, 2} s1.symmetric_difference(s2) # {1, 2, 4, 5} # Modification s1.add(4) # Add element s1.update([5, 6]) # Add multiple s1.discard(3) # Remove (no error) s1.remove(2) # Remove (KeyError if missing) # Check methods s1.issubset(s2) # False s1.issuperset(s2) # False s1.isdisjoint(s2) # False (share element 3)

Instance Methods Basic

Instance methods are the most common type of methods in Python. They are bound to the instance of a class and can access and modify object state.

class Person: def __init__(self, name, age): self.name = name # Instance attribute self.age = age # Instance attribute # Instance method - can access self def introduce(self): return f"Hi, I'm {self.name}, {self.age} years old" # Instance method that modifies state def have_birthday(self): self.age += 1 return f"Happy Birthday! Now I'm {self.age}" # Instance method with additional parameters def greet(self, other_name): return f"Hello {other_name}, I'm {self.name}" # Usage person = Person("Alice", 30) print(person.introduce()) # Hi, I'm Alice, 30 years old print(person.have_birthday()) # Happy Birthday! Now I'm 31 print(person.greet("Bob")) # Hello Bob, I'm Alice

Key Point: The first parameter of an instance method is always self, which refers to the instance calling the method. Python passes this automatically.

Class Methods Intermediate

Class methods are bound to the class rather than instances. They receive the class as the first parameter (conventionally named cls) and can modify class state.

class Circle: pi = 3.14159 # Class attribute def __init__(self, radius): self.radius = radius # Class method as alternative constructor @classmethod def from_diameter(cls, diameter): return cls(diameter / 2) # Class method to modify class attribute @classmethod def set_pi(cls, value): cls.pi = value # Class method for factory pattern @classmethod def unit_circle(cls): return cls(1) def area(self): return self.pi * self.radius ** 2 # Usage - call on class, not instance c1 = Circle.from_diameter(10) # radius=5 c2 = Circle.unit_circle() # radius=1 Circle.set_pi(3.14) # Modify class attribute

Use Cases for Class Methods

  • Alternative Constructors: Factory methods with different initialization
  • Class State Modification: Update class-level attributes
  • Factory Patterns: Create instances in different ways
  • Inheritance Friendly: cls refers to the actual class being called

@classmethod vs @staticmethod

# @classmethod: receives cls @classmethod def create(cls, data): return cls(data) # Can create instances # @staticmethod: no cls or self @staticmethod def validate(data): return data is not None

Static Methods Intermediate

Static methods don't receive any special first argument. They behave like regular functions but belong to the class's namespace for organizational purposes.

class MathUtils: @staticmethod def add(a, b): """Add two numbers.""" return a + b @staticmethod def is_even(n): """Check if a number is even.""" return n % 2 == 0 @staticmethod def factorial(n): """Calculate factorial recursively.""" if n <= 1: return 1 return n * MathUtils.factorial(n - 1) @staticmethod def clamp(value, min_val, max_val): """Clamp value between min and max.""" return max(min_val, min(max_val, value)) # Usage - can call on class or instance print(MathUtils.add(5, 3)) # 8 print(MathUtils.is_even(4)) # True print(MathUtils.factorial(5)) # 120 print(MathUtils.clamp(15, 0, 10)) # 10 # Can also call from instance (not recommended) m = MathUtils() print(m.add(1, 2)) # 3

When to Use Static Methods

  • Utility functions related to the class
  • No access to class or instance needed
  • Grouping related functionality together
  • Alternative to module-level functions

When NOT to Use Static Methods

  • When you need access to class attributes
  • When you need access to instance attributes
  • When the function could be a standalone function
  • When inheritance behavior matters

Property Methods Intermediate

Properties provide a way to define methods that can be accessed like attributes, enabling controlled attribute access with getters, setters, and deleters.

class Temperature: def __init__(self, celsius=0): self._celsius = celsius # Private backing field # Getter - accessed like an attribute @property def celsius(self): return self._celsius # Setter - called when assigning @celsius.setter def celsius(self, value): if value < -273.15: raise ValueError("Temperature below absolute zero!") self._celsius = value # Computed property (read-only) @property def fahrenheit(self): return (self._celsius * 9/5) + 32 @fahrenheit.setter def fahrenheit(self, value): self.celsius = (value - 32) * 5/9 # Deleter @celsius.deleter def celsius(self): self._celsius = 0 # Usage - accessed like attributes, not methods temp = Temperature(25) print(temp.celsius) # 25 (getter) print(temp.fahrenheit) # 77.0 (computed property) temp.celsius = 30 # setter called temp.fahrenheit = 100 # setter converts del temp.celsius # deleter called

Warning: Use underscore prefix (_celsius) for backing fields to indicate they should not be accessed directly. This is a Python convention, not true encapsulation.

Magic Methods (Dunder Methods) Advanced

Magic methods (double underscore or "dunder" methods) are special methods that Python calls behind the scenes. They allow your objects to work with Python's built-in functions and operators.

Common Magic Methods

  • __init__ - Constructor
  • __str__ - String representation (print)
  • __repr__ - Official representation
  • __len__ - Length (len())
  • __getitem__ - Index access
  • __eq__ - Equality (==)

Why Use Magic Methods?

  • Make objects behave like built-in types
  • Enable Pythonic syntax and patterns
  • Integrate with Python's data model
  • Create intuitive APIs for your classes

Tip: Never call magic methods directly (e.g., obj.__str__()). Instead, use the corresponding built-in function (e.g., str(obj)). Python calls them automatically.

__init__ and __new__ Advanced

__new__ creates the instance, while __init__ initializes it. Understanding both is crucial for advanced object creation patterns.

class Singleton: _instance = None def __new__(cls, *args, **kwargs): """Create the instance (called first).""" if cls._instance is None: cls._instance = super().__new__(cls) return cls._instance def __init__(self, value=0): """Initialize the instance (called second).""" self.value = value # __new__ creates, __init__ initializes s1 = Singleton(10) s2 = Singleton(20) print(s1 is s2) # True (same instance) print(s1.value) # 20 (re-initialized) class ImmutablePoint: def __new__(cls, x, y): # Validate before creating if x < 0 or y < 0: raise ValueError("Coordinates must be positive") return super().__new__(cls) def __init__(self, x, y): self.x = x self.y = y

When to Use __new__

  • Singleton pattern implementation
  • Immutable object creation
  • Subclassing built-in immutable types
  • Object caching/pooling
  • Return different types conditionally

__init__ vs __new__

  • __new__ is a class method, __init__ is instance method
  • __new__ returns the instance, __init__ returns None
  • __new__ called once, __init__ called every instantiation
  • Override __new__ only when necessary

Operator Overloading Advanced

Override operators like +, -, ==, and others by defining corresponding magic methods.

class Vector: def __init__(self, x, y): self.x = x self.y = y # Addition: v1 + v2 def __add__(self, other): return Vector(self.x + other.x, self.y + other.y) # Subtraction: v1 - v2 def __sub__(self, other): return Vector(self.x - other.x, self.y - other.y) # Multiplication: v * scalar def __mul__(self, scalar): return Vector(self.x * scalar, self.y * scalar) # Reverse multiplication: scalar * v def __rmul__(self, scalar): return self.__mul__(scalar) # Equality: v1 == v2 def __eq__(self, other): return self.x == other.x and self.y == other.y # Negation: -v def __neg__(self): return Vector(-self.x, -self.y) # Absolute (magnitude): abs(v) def __abs__(self): return (self.x**2 + self.y**2) ** 0.5 # Usage v1 = Vector(1, 2) v2 = Vector(3, 4) print(v1 + v2) # Vector(4, 6) print(v2 - v1) # Vector(2, 2) print(v1 * 3) # Vector(3, 6) print(3 * v1) # Vector(3, 6) - uses __rmul__ print(v1 == v2) # False print(abs(v1)) # 2.236...

Arithmetic Operators

  • __add__ - Addition (+)
  • __sub__ - Subtraction (-)
  • __mul__ - Multiplication (*)
  • __truediv__ - Division (/)
  • __floordiv__ - Floor division (//)
  • __mod__ - Modulo (%)
  • __pow__ - Power (**)

Comparison Operators

  • __eq__ - Equal (==)
  • __ne__ - Not equal (!=)
  • __lt__ - Less than (<)
  • __le__ - Less or equal (<=)
  • __gt__ - Greater than (>)
  • __ge__ - Greater or equal (>=)

String Representation Advanced

Control how objects are displayed as strings using __str__ and __repr__.

class Book: def __init__(self, title, author, pages): self.title = title self.author = author self.pages = pages # __str__ - User-friendly representation # Called by print(), str(), format() def __str__(self): return f"'{self.title}' by {self.author}" # __repr__ - Developer-friendly representation # Called by repr(), in debugger, list display def __repr__(self): return f"Book(title='{self.title}', author='{self.author}', pages={self.pages})" # Usage book = Book("Python 101", "John Doe", 350) print(str(book)) # 'Python 101' by John Doe print(repr(book)) # Book(title='Python 101', author='John Doe', pages=350) # In a list, __repr__ is shown books = [book] print(books) # [Book(title='Python 101', ...)] # f-string uses __str__ first, then __repr__ print(f"Book: {book}") # Book: 'Python 101' by John Doe

__str__ Guidelines

  • User-friendly and readable
  • Should return a string
  • Focus on clarity over completeness
  • Like a label or display name

__repr__ Guidelines

  • Developer-friendly and unambiguous
  • Should ideally be valid Python code
  • Include all important attributes
  • Like a constructor call

Container Methods Advanced

Make your objects behave like containers (lists, dicts) by implementing container protocol methods.

class Inventory: def __init__(self): self.items = {} # len(inventory) def __len__(self): return len(self.items) # inventory[key] def __getitem__(self, key): return self.items[key] # inventory[key] = value def __setitem__(self, key, value): self.items[key] = value # del inventory[key] def __delitem__(self, key): del self.items[key] # key in inventory def __contains__(self, key): return key in self.items # String representation def __repr__(self): return f"Inventory({self.items})" # Usage - behaves like a dict inv = Inventory() inv["apple"] = 10 inv["banana"] = 20 print(len(inv)) # 2 print(inv["apple"]) # 10 print("apple" in inv) # True del inv["banana"] print(inv) # Inventory({'apple': 10})

Sequence Methods

  • __len__ - len(obj)
  • __getitem__ - obj[key]
  • __setitem__ - obj[key] = val
  • __delitem__ - del obj[key]
  • __contains__ - item in obj

Additional Methods

  • __iter__ - for x in obj
  • __reversed__ - reversed(obj)
  • __count__ - obj.count(item)
  • __index__ - obj.index(item)

Iteration Methods Advanced

Make your objects iterable by implementing __iter__ and __next__.

# Iterable - has __iter__ that returns an iterator class CountUp: def __init__(self, start, end): self.start = start self.end = end def __iter__(self): return CountUpIterator(self.start, self.end) class CountUpIterator: def __init__(self, start, end): self.current = start self.end = end def __iter__(self): return self def __next__(self): if self.current > self.end: raise StopIteration value = self.current self.current += 1 return value # Usage for num in CountUp(1, 5): print(num) # 1, 2, 3, 4, 5 # Simpler approach with generator class CountUp2: def __init__(self, start, end): self.start = start self.end = end def __iter__(self): for i in range(self.start, self.end + 1): yield i

Iterable vs Iterator

  • Iterable: Has __iter__(), returns an iterator
  • Iterator: Has __next__(), returns values
  • Iterators are also iterables (return self)
  • Can only iterate once with iterators

Generator Shortcut

  • Use yield instead of implementing __iter__ and __next__
  • Automatic StopIteration handling
  • Memory efficient for large sequences
  • Can be composed and chained

Abstract Methods Advanced

Abstract methods define a contract that subclasses must implement. Use the abc module to create abstract base classes.

from abc import ABC, abstractmethod class Shape(ABC): """Abstract base class for shapes.""" @abstractmethod def area(self): """Calculate the area - must be implemented.""" pass @abstractmethod def perimeter(self): """Calculate the perimeter - must be implemented.""" pass # Concrete method (optional implementation) def description(self): return f"{self.__class__.__name__}: area={self.area():.2f}" class Circle(Shape): def __init__(self, radius): self.radius = radius def area(self): return 3.14159 * self.radius ** 2 def perimeter(self): return 2 * 3.14159 * self.radius class Rectangle(Shape): def __init__(self, width, height): self.width = width self.height = height def area(self): return self.width * self.height def perimeter(self): return 2 * (self.width + self.height) # Cannot instantiate abstract class # shape = Shape() # TypeError! # Must implement all abstract methods circle = Circle(5) print(circle.area()) # 78.53975 print(circle.description()) # Circle: area=78.54

Methods as Decorators Advanced

Methods can be used as decorators to modify or wrap other functions and methods.

class Timer: def __init__(self, func): self.func = func self.calls = 0 def __call__(self, *args, **kwargs): self.calls += 1 import time start = time.perf_counter() result = self.func(*args, **kwargs) elapsed = time.perf_counter() - start print(f"{self.func.__name__} took {elapsed:.4f}s") return result @Timer def slow_function(): import time time.sleep(1) return "Done" slow_function() # Prints: slow_function took 1.0012s slow_function() # Prints: slow_function took 1.0015s
class LoggingMixin: """Mixin that adds logging to methods.""" def log_call(self, func): def wrapper(*args, **kwargs): print(f"Calling {func.__name__}") result = func(*args, **kwargs) print(f"Finished {func.__name__}") return result return wrapper class MyService(LoggingMixin): def process_data(self, data): return [x * 2 for x in data] # Method as decorator in action service = MyService() # Would need to be applied: service.process_data = service.log_call(service.process_data)

NumPy Methods Intermediate

Essential NumPy array methods for numerical computing.

Array Creation

import numpy as np # Create arrays np.array([1, 2, 3]) np.zeros((3, 4)) np.ones((2, 3)) np.arange(0, 10, 2) np.linspace(0, 1, 100) np.random.rand(3, 3)

Array Attributes

arr = np.array([[1,2], [3,4]]) arr.shape # (2, 2) arr.ndim # 2 arr.size # 4 arr.dtype # int64 arr.itemsize # 8 bytes

Mathematical Operations

a = np.array([1, 2, 3]) b = np.array([4, 5, 6]) # Element-wise operations a + b # [5, 7, 9] a * b # [4, 10, 18] np.sqrt(a) # [1, 1.414, 1.732] np.exp(a) # [2.718, 7.389, 20.086] np.log(a) # [0, 0.693, 1.099]

Aggregation Methods

arr = np.array([1, 2, 3, 4, 5]) arr.sum() # 15 arr.mean() # 5.0 arr.std() # 1.414 arr.min() # 1 arr.max() # 5 arr.argmax() # 4 (index of max) # Axis operations matrix = np.array([[1,2], [3,4]]) matrix.sum(axis=0) # [4, 6] matrix.sum(axis=1) # [3, 7]
# Reshaping methods arr = np.arange(12) arr.reshape(3, 4) # 3x4 matrix arr.reshape(2, 2, 3) # 2x2x3 tensor arr.flatten() # 1D array arr.ravel() # 1D array (view) arr.T # Transpose # Indexing methods arr = np.array([10, 20, 30, 40, 50]) arr.take([0, 2, 4]) # [10, 30, 50] arr.put([0, 2], [100, 300]) # Modify at indices Copy

Tip: NumPy methods are implemented in C and are significantly faster than Python loops. Always prefer vectorized operations over loops for numerical computations.
