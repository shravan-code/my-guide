Pandas Methods

A focused guide to Pandas Methods.

Series Creation Methods Basic

Methods for creating Series objects from various data sources.

import pandas as pd import numpy as np # pd.Series() - Create from various sources s1 = pd.Series([1, 2, 3]) # From list s2 = pd.Series({'a': 1, 'b': 2}) # From dict s3 = pd.Series(np.array([1, 2, 3])) # From NumPy array s4 = pd.Series(5, index=['a', 'b', 'c']) # From scalar s5 = pd.Series([1, 2, 3], dtype='float64') # With dtype s6 = pd.Series([1, 2, 3], name='my_series') # With name # pd.date_range() - Create datetime index dates = pd.date_range('2024-01-01', periods=10, freq='D') ts = pd.Series(np.random.randn(10), index=dates) # pd.bdate_range() - Business day range business_dates = pd.bdate_range('2024-01-01', periods=10)

Tip: Use dtype parameter to specify data type and save memory. For example, dtype='int32' uses less memory than int64.

Series Access & Indexing Methods Basic

Methods to access and index Series elements.

Index Access

s = pd.Series([10, 20, 30, 40], index=['a', 'b', 'c', 'd']) # .loc[] - Label-based s.loc['a'] # 10 s.loc[['a', 'c']] # Series with a, c s.loc['a':'c'] # Slice by label # .iloc[] - Integer position s.iloc[0] # 10 s.iloc[[0, 2]] # Series at positions 0, 2 s.iloc[1:3] # Slice by position

Query Methods

# .get() - Safe access with default s.get('a') # 10 s.get('z', 0) # 0 (default) # .at[] / .iat[] - Fast scalar access s.at['a'] # 10 s.iat[0] # 10 # .head() / .tail() s.head(2) # First 2 elements s.tail(2) # Last 2 elements # .idxmin() / .idxmax() - Index of min/max s.idxmin() # 'a' (index of minimum) s.idxmax() # 'd' (index of maximum)

Series Sorting Methods Basic

Methods to sort Series by values or index.

Series Missing Data Methods Intermediate

Methods to detect, handle, and fill missing values.

Detection

s = pd.Series([1, 2, np.nan, 4, np.nan]) # Detection methods s.isna() # Boolean: [F, F, T, F, T] s.isnull() # Same as isna() s.notna() # Boolean: [T, T, F, T, F] s.notnull() # Same as notna() s.hasnans() # True if any NaN s.count() # 3 (non-null count)

Removal

# .dropna() - Remove missing values s.dropna() # Remove NaN entries s.dropna(how='all') # Only drop if all NaN # .replace() - Replace values s.replace(np.nan, 0) # Replace NaN with 0 s.replace({np.nan: 0}) # Dict replacement s.replace([1, 2], 99) # Replace multiple
# .fillna() - Fill missing values s.fillna(0) # Fill with 0 s.fillna(s.mean()) # Fill with mean s.fillna(s.median()) # Fill with median s.fillna(method='ffill') # Forward fill s.fillna(method='bfill') # Backward fill s.ffill() # Forward fill (shorthand) s.bfill() # Backward fill (shorthand) # .interpolate() - Fill with interpolation s.interpolate() # Linear (default) s.interpolate(method='spline') # Spline interpolation s.interpolate(method='time') # Time-weighted s.interpolate(limit=2) # Limit fill length s.interpolate(limit_direction='both') # Fill both directions

Series String Methods Intermediate

Access string methods via the .str accessor for Series with string data.

Case & Whitespace

s = pd.Series(['Hello', 'WORLD', ' test ']) s.str.lower() # ['hello', 'world', ' test '] s.str.upper() # ['HELLO', 'WORLD', ' TEST '] s.str.title() # ['Hello', 'World', ' Test '] s.str.capitalize() # ['Hello', 'World', ' test '] s.str.swapcase() # ['hELLO', 'world', ' TEST '] s.str.strip() # ['Hello', 'WORLD', 'test'] s.str.lstrip() # Left strip s.str.rstrip() # Right strip

Search & Find

s = pd.Series(['apple', 'banana', 'cherry']) s.str.contains('a') # [True, True, False] s.str.startswith('a') # [True, False, False] s.str.endswith('e') # [True, False, True] s.str.find('a') # [1, 1, -1] s.str.count('a') # [1, 3, 0] s.str.len() # [5, 6, 6] s.str.match(r'^[a-c]') # Regex match

Transform

# Replace s.str.replace('a', 'X') s.str.replace(r'[aeiou]', '*', regex=True) # Split and join s.str.split('a') # Split by 'a' s.str.split(expand=True) # Expand to columns s.str.join('-') # Join characters # Slice s.str[0] # First character s.str[1:3] # Characters 1-3

Extract & Encode

# Extract regex groups s = pd.Series(['John 25', 'Jane 30']) s.str.extract(r'(\w+) (\d+)') # Returns DataFrame with columns [0, 1] # Extract named groups s.str.extract(r'(?P\w+) (?P\d+)') # Get dummies (one-hot encode) pd.get_dummies(s.str.split(' ').explode()) # Encoding s.str.encode('utf-8') # Encode to bytes s.str.decode('utf-8') # Decode from bytes

# Date arithmetic and transformations dates.dt.strftime('%Y-%m-%d') # Format to string dates.dt.to_period('M') # Convert to period dates.dt.to_pydatetime() # Convert to datetime dates.dt.tz_localize('UTC') # Add timezone dates.dt.tz_convert('US/Eastern') # Convert timezone # Timedelta operations timedeltas = pd.Series(pd.to_timedelta(['1 days', '2 days'])) timedeltas.dt.days # Days component timedeltas.dt.seconds # Seconds component timedeltas.dt.total_seconds() # Total seconds

Split data into groups and apply aggregate functions.

DataFrame Merge & Join Methods Intermediate

Combine multiple DataFrames using various join strategies.

# pd.merge() - SQL-style joins pd.merge(df1, df2, on='key') # Inner join pd.merge(df1, df2, on='key', how='left') # Left join pd.merge(df1, df2, on='key', how='right') # Right join pd.merge(df1, df2, on='key', how='outer') # Full outer pd.merge(df1, df2, on='key', how='cross') # Cross join # Different column names pd.merge(df1, df2, left_on='user_id', right_on='id') # Multiple keys pd.merge(df1, df2, on=['key1', 'key2']) # Suffixes for duplicate columns pd.merge(df1, df2, on='key', suffixes=('_left', '_right')) # Index-based merge pd.merge(df1, df2, left_index=True, right_index=True) # df.join() - Join on index df1.join(df2) # Left join on index df1.join(df2, how='outer') # Full outer join # pd.concat() - Stack DataFrames pd.concat([df1, df2]) # Stack rows pd.concat([df1, df2], axis=1) # Stack columns pd.concat([df1, df2], ignore_index=True) # Reset index

Merge Parameters

  • on - Column name(s) to join on
  • how - Join type (inner, left, right, outer)
  • left_on/right_on - Column names for left/right
  • suffixes - Suffixes for duplicate columns
  • validate - Check merge validity

Join Types Explained

  • inner - Only matching rows
  • left - All left, matching right
  • right - All right, matching left
  • outer - All rows from both
  • cross - Cartesian product

DataFrame Apply & Transform Methods Intermediate

Apply functions to DataFrames for custom transformations.

# .apply() - Apply function to rows/columns df.apply(lambda x: x.max() - x.min()) # Column-wise df.apply(lambda x: x.max() - x.min(), axis=1) # Row-wise df.apply(np.sum) # Apply numpy function df.apply(lambda x: x.quantile(0.25)) # Custom function # .applymap() / .map() - Element-wise df.map(lambda x: x.strip() if isinstance(x, str) else x) df.map('{:.2f}'.format) # Format numbers # .pipe() - Method chaining with functions def add_total(df): return df.assign(total=df.sum(axis=1)) def filter_positive(df): return df[df['total'] > 0] result = df.pipe(add_total).pipe(filter_positive) # .assign() - Create new columns df.assign( bmi=lambda x: x['weight'] / (x['height'] ** 2), category=lambda x: pd.cut(x['bmi'], bins=[0, 18.5, 25, 100]) )

Warning: .apply() is significantly slower than vectorized operations. Use built-in methods and vectorized operations whenever possible for better performance.

DataFrame Window Functions Advanced

Apply rolling, expanding, and exponentially weighted calculations.

# .rolling() - Rolling window calculations df['MA7'] = df['close'].rolling(window=7).mean() df['MA30'] = df['close'].rolling(30).mean() df['std'] = df['close'].rolling(20).std() # Rolling with multiple aggregations df['rolling_stats'] = df['value'].rolling(5).agg(['mean', 'std', 'min', 'max']) # Rolling apply with custom function df['custom'] = df['value'].rolling(3).apply(lambda x: x.max() - x.min()) # min_periods - Minimum observations required df['MA5'] = df['close'].rolling(5, min_periods=1).mean()

Expanding Windows

# .expanding() - Cumulative from start df['exp_mean'] = df['value'].expanding().mean() df['exp_max'] = df['value'].expanding().max() df['exp_std'] = df['value'].expanding().std() # Equivalent to cumsum/cummean df['cummean'] = df['value'].expanding().mean()

Exponentially Weighted

# .ewm() - Exponentially weighted df['ewm_mean'] = df['value'].ewm(span=10).mean() df['ewm_std'] = df['value'].ewm(span=10).std() # span, com, halflife, alpha df['ewma'] = df['value'].ewm(alpha=0.3).mean() # Adjust for bias df['ewm_adj'] = df['value'].ewm(span=10, adjust=True).mean()
# .shift() with groupby - Lag features df['prev_value'] = df.groupby('category')['value'].shift(1) # Differencing within groups df['diff'] = df.groupby('category')['value'].diff() # Rank within groups df['rank'] = df.groupby('category')['value'].rank(ascending=False)

Tip: Use min_periods=1 in rolling windows to avoid NaN values at the start. For EWMA, use span parameter which controls the decay rate (larger span = smoother).
