Data Structures Basic

Tables

Tables

Primary structured storage in PostgreSQL.

CREATE TABLE users ( id SERIAL PRIMARY KEY, username VARCHAR(50) UNIQUE NOT NULL, email VARCHAR(255) NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Insert data INSERT INTO users (username, email) VALUES ('alice', 'alice@example.com');
Arrays

Arrays

Built‑in array data type for storing lists.

CREATE TABLE products ( id INT PRIMARY KEY, name TEXT, tags TEXT[] ); -- Insert with array INSERT INTO products VALUES (1, 'Laptop', ARRAY['electronics', 'computers']); -- Query array SELECT * FROM products WHERE 'electronics' = ANY(tags);

Window Functions Advanced

Ranking

ROW_NUMBER, RANK, DENSE_RANK

Rank rows within partitions.

SELECT id, name, amount, ROW_NUMBER() OVER w as rn, RANK() OVER w as rnk, DENSE_RANK() OVER w as dense_rnk FROM orders WINDOW w AS ( PARTITION BY category ORDER BY amount DESC );
Offset

LAG & LEAD

Access previous/next rows.

SELECT date, amount, LAG(amount, 1) OVER (ORDER BY date) as prev_amount, LEAD(amount, 1) OVER (ORDER BY date) as next_amount, amount - LAG(amount, 1) OVER (ORDER BY date) as diff FROM daily_sales;
Aggregation

Running Totals

Cumulative aggregations.

SELECT date, amount, SUM(amount) OVER (ORDER BY date) as running_total, AVG(amount) OVER ( ORDER BY date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW ) as moving_avg_7d, COUNT(*) OVER (PARTITION BY MONTH(date)) as monthly_count FROM daily_sales;
NTILE

NTILE & PERCENTILE

Divide into buckets.

SELECT id, amount, NTILE(4) OVER (ORDER BY amount) as quartile, PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY amount) as median, PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY amount) as p25, PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY amount) as p75 FROM orders;

CTEs & Recursive Queries Advanced

CTE

Common Table Expressions

Create reusable query blocks.

WITH monthly_sales AS ( SELECT DATE_TRUNC('month', created_at) as month, SUM(amount) as total, COUNT(*) as cnt FROM orders GROUP BY 1 ), growth AS ( SELECT month, total, LAG(total) OVER (ORDER BY month) as prev_month, ROUND((total - LAG(total) OVER (ORDER BY month)) / LAG(total) OVER (ORDER BY month) * 100, 2) as growth_pct FROM monthly_sales ) SELECT * FROM growth ORDER BY month;
Recursive

Recursive CTE

Query hierarchical data.

WITH RECURSIVE org_chart AS ( -- Base case: top-level manager SELECT id, name, manager_id, 1 as level FROM employees WHERE manager_id IS NULL UNION ALL -- Recursive case: employees under managers SELECT e.id, e.name, e.manager_id, oc.level + 1 FROM employees e JOIN org_chart oc ON e.manager_id = oc.id ) SELECT * FROM org_chart ORDER BY level, id;
Multiple CTEs

Chained CTEs

Use multiple CTEs together.

WITH active_users AS ( SELECT id, name, email FROM users WHERE status = 'active' ), user_orders AS ( SELECT user_id, COUNT(*) as order_count, SUM(amount) as total_spent FROM orders GROUP BY user_id ) SELECT u.name, u.email, COALESCE(o.order_count, 0) as orders, COALESCE(o.total_spent, 0) as spent FROM active_users u LEFT JOIN user_orders o ON u.id = o.user_id;
Materialized

Materialized CTE

Force CTE materialization.

WITH large_cte AS MATERIALIZED ( SELECT * FROM huge_table WHERE complex_condition = true ) SELECT a.id, (SELECT COUNT(*) FROM large_cte WHERE category = a.category) as cat_count FROM another_table a; -- Or use subquery with LATERAL SELECT u.id, stats.* FROM users u CROSS JOIN LATERAL ( SELECT COUNT(*) as order_count, SUM(amount) as total FROM orders o WHERE o.user_id = u.id ) stats;

Aggregation Medium

GROUPING SETS

Multiple Aggregation Levels

Aggregate at multiple levels in one query.

SELECT COALESCE(category, 'ALL') as category, COALESCE(status, 'ALL') as status, SUM(amount) as total, COUNT(*) as cnt FROM orders GROUP BY GROUPING SETS ( (category, status), -- Both (category), -- By category only (status), -- By status only () -- Grand total );
ROLLUP

ROLLUP

Generate subtotals and grand total.

SELECT region, product, SUM(sales) as total_sales FROM sales_data GROUP BY ROLLUP(region, product); -- Equivalent to GROUPING SETS: -- (region, product), (region), ()
CUBE

CUBE

All combinations of grouping columns.

SELECT region, product, YEAR(sale_date) as year, SUM(sales) as total_sales FROM sales_data GROUP BY CUBE(region, product, YEAR(sale_date)); -- Generates 2^n combinations: -- (r,p,y), (r,p), (r,y), (p,y), (r), (p), (y), ()
FILTER

Conditional Aggregation

Aggregate with FILTER clause.

SELECT category, COUNT(*) as total, COUNT(*) FILTER (WHERE status = 'active') as active, COUNT(*) FILTER (WHERE status = 'inactive') as inactive, SUM(amount) FILTER (WHERE created_at >= NOW() - INTERVAL '30 days') as last_30d, AVG(amount) FILTER (WHERE amount > 100) as avg_large FROM orders GROUP BY category;

JSON Operations Medium

Extract

JSON Extraction

Extract values from JSON columns.

-- Arrow operator (returns JSONB value) data->>'name' -- text data->'address'->>'city' -- text nested -- Arrow with integer index data->0 -- first array element data->'tags'->1 -- second tag -- jsonb_extract_path (alternative) jsonb_extract_path(data, 'address', 'city') -- Multiple values SELECT data->>'id' as id, data->>'name' as name, data->'metadata'->>'created_at' as created FROM users;
Build

Build JSON

Create JSON from query results.

-- Single object SELECT jsonb_build_object( 'id', id, 'name', name, 'total', amount ) as result FROM orders; -- Array of objects SELECT jsonb_agg( jsonb_build_object('id', id, 'name', name) ) as users_array FROM users; -- Nested structure SELECT jsonb_build_object( 'category', category, 'orders', jsonb_agg( jsonb_build_object('id', id, 'amount', amount) ) ) FROM orders GROUP BY category;
Query

Query JSON

Filter and search JSON data.

-- Existence check data ? 'name' -- key exists data ?| array['a','b'] -- any key exists data ?& array['a','b'] -- all keys exist -- Containment data @> '{"status": "active"}'::jsonb -- Like matching on text values data->>'name' LIKE '%test%' -- Filter SELECT * FROM events WHERE data->>'type' = 'purchase' AND (data->>'amount')::numeric > 100;
Index

JSON Indexing

Index JSON for performance.

-- GIN index for containment queries CREATE INDEX idx_data_gin ON table_name USING GIN (data); -- Expression index on specific key CREATE INDEX idx_data_status ON table_name ((data->>'status')); -- Partial index on JSON CREATE INDEX idx_active_users ON users ((data->>'email')) WHERE data->>'status' = 'active'; -- BRIN for time-series JSON CREATE INDEX idx_events_time ON events USING BRIN ((data->>'timestamp'));

Upsert & Manipulation Medium

UPSERT

INSERT ON CONFLICT

Insert or update based on conflicts.

INSERT INTO users (id, name, email, updated_at) VALUES (1, 'John', 'john@example.com', NOW()) ON CONFLICT (id) DO UPDATE SET name = EXCLUDED.name, email = EXCLUDED.email, updated_at = EXCLUDED.updated_at; -- Do nothing on conflict INSERT INTO users (id, name) VALUES (1, 'John') ON CONFLICT (id) DO NOTHING; -- Partial unique index CREATE UNIQUE INDEX idx_active_email ON users (email) WHERE status = 'active';
MERGE

MERGE Statement

SQL:2003 standard merge (UPSERT).

MERGE INTO target t USING source s ON t.id = s.id WHEN MATCHED AND s.deleted = true THEN DELETE WHEN MATCHED THEN UPDATE SET name = s.name, email = s.email, updated_at = NOW() WHEN NOT MATCHED THEN INSERT (id, name, email, created_at) VALUES (s.id, s.name, s.email, NOW());
DELETE

DELETE with RETURNING

Delete and return affected rows.

-- Delete and return DELETE FROM orders WHERE created_at < NOW() - INTERVAL '1 year' RETURNING id, name, amount; -- Delete with CTE WITH old_records AS ( SELECT id FROM archive_logs WHERE created_at < NOW() - INTERVAL '90 days' ) DELETE FROM logs WHERE id IN (SELECT id FROM old_records) RETURNING *; -- Truncate with cascade TRUNCATE TABLE logs CASCADE;
UPDATE

UPDATE with FROM

Update using joins.

-- Update with JOIN UPDATE orders o SET status = 'processed' FROM customers c WHERE o.customer_id = c.id AND c.premium = true AND o.status = 'pending'; -- Update with subquery UPDATE products p SET price = price * 1.1 WHERE category_id IN ( SELECT id FROM categories WHERE name = 'Electronics' ); -- Update from CTE WITH stats AS ( SELECT user_id, SUM(amount) as total FROM orders GROUP BY user_id ) UPDATE users u SET total_spent = stats.total FROM stats WHERE u.id = stats.user_id;

Performance Advanced

EXPLAIN

Query Analysis

Analyze query execution plans.

-- Basic explain EXPLAIN SELECT * FROM orders WHERE user_id = 123; -- With actual execution EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT) SELECT * FROM orders WHERE user_id = 123; -- Key metrics to check: -- - Seq Scan vs Index Scan -- - Rows removed by filter -- - Actual vs Estimated rows -- - Buffers (shared hit/read) -- - Execution time
Indexes

Index Types

Choose the right index type.

-- B-tree (default, ordered) CREATE INDEX idx_orders_user ON orders(user_id); -- Composite index CREATE INDEX idx_orders_user_date ON orders(user_id, created_at DESC); -- Partial index CREATE INDEX idx_active_orders ON orders(created_at) WHERE status = 'active'; -- Covering index (includes) CREATE INDEX idx_orders_covering ON orders(user_id) INCLUDE (amount, status); -- BRIN for large sorted data CREATE INDEX idx_events_brin ON events USING BRIN (created_at);
Vacuum

Maintenance

Database maintenance commands.

-- Analyze table stats ANALYZE orders; -- Vacuum (reclaim space) VACUUM orders; VACUUM FULL orders; -- More aggressive -- Reindex REINDEX TABLE orders; REINDEX INDEX idx_orders_user; -- Update statistics ANALYZE VERBOSE orders; -- Check table bloat SELECT schemaname, tablename, n_live_tup, n_dead_tup, round(100*n_dead_tup/(n_live_tup+n_dead_tup+1),1) as dead_pct FROM pg_stat_user_tables ORDER BY n_dead_tup DESC;
Monitoring

Performance Monitoring

Find slow queries and issues.

-- Current queries SELECT pid, query, state, duration_ms FROM pg_stat_activity WHERE state = 'active' ORDER BY duration_ms DESC; -- Slow queries (requires pg_stat_statements) SELECT query, calls, mean_exec_time, total_exec_time FROM pg_stat_statements ORDER BY mean_exec_time DESC LIMIT 10; -- Table sizes SELECT schemaname, tablename, pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) as size FROM pg_tables ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC;

Tips & Tricks

Tip

Use EXPLAIN ANALYZE for query optimization

Understand query execution plans and identify bottlenecks.

EXPLAIN ANALYZE SELECT * FROM orders WHERE customer_id = 123 ORDER BY created_at DESC LIMIT 10;
Trick

Use CTEs for complex queries

Break down complex logic into readable steps.

WITH monthly_sales AS ( SELECT DATE_TRUNC('month', order_date) as month, SUM(total_amount) as total FROM orders GROUP BY 1 ), growth AS ( SELECT month, total, LAG(total) OVER (ORDER BY month) as prev_total FROM monthly_sales ) SELECT * FROM growth;
Tip

Use window functions for analytics

Perform calculations across rows without grouping.

SELECT customer_id, order_date, total_amount, SUM(total_amount) OVER ( PARTITION BY customer_id ORDER BY order_date ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW ) as running_total, ROW_NUMBER() OVER ( PARTITION BY customer_id ORDER BY order_date DESC ) as rn FROM orders;

Best Practices Medium

Indexing

Use the right indexes

Index columns used in joins and where clauses to speed lookups.

CREATE INDEX idx_orders_customer ON orders(customer_id); CREATE INDEX idx_orders_date ON orders(order_date);
Vacuum

Regular maintenance

Run VACUUM and ANALYZE to keep optimizer stats fresh and reduce bloat.

VACUUM ANALYZE; ANALYZE orders;
Transactions

Use explicit transactions

Group DML in transactions to maintain data integrity.

BEGIN; UPDATE accounts SET balance = balance - 100 WHERE id = 1; UPDATE accounts SET balance = balance + 100 WHERE id = 2; COMMIT;
