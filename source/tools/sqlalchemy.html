SQLAlchemy

Complete Guide with Python Examples

Introduction

SQLAlchemy is the most popular and feature-rich Python SQL toolkit and Object Relational Mapper (ORM). It provides a full suite of well-known enterprise-level persistence patterns, designed for efficient and high-performing database access.

SQLAlchemy consists of two distinct components:

check_circle

SQLAlchemy Core

A schema-centric, SQL-expression toolkit

check_circle

SQLAlchemy ORM

An object-centric, domain-model layer built on top of Core

Engine

The Engine is the starting point for any SQLAlchemy application.

from sqlalchemy import create_engine # SQLite (file-based) engine = create_engine("sqlite:///mydb.db", echo=True) # PostgreSQL engine = create_engine( "postgresql+psycopg2://user:password@localhost:5432/mydb", pool_size=5, max_overflow=10 ) # In-memory SQLite (for testing) engine = create_engine("sqlite:///:memory:")

Session

The Session is the primary interface for persistence operations in SQLAlchemy ORM.

from sqlalchemy.orm import sessionmaker, Session # Method 1: sessionmaker factory SessionLocal = sessionmaker(autocommit=False, autoflush=True, bind=engine) session = SessionLocal() try: session.add(user) session.commit() except Exception: session.rollback() raise finally: session.close() # Method 2: Context manager (SQLAlchemy 1.4+) with Session(engine) as session: session.add(user) session.commit()

Declarative Base

The Declarative system allows you to define mapped classes and Table metadata together.

from sqlalchemy import Column, Integer, String, DateTime, Boolean, Float from sqlalchemy.orm import DeclarativeBase from datetime import datetime class Base(DeclarativeBase): pass class User(Base): __tablename__ = "users" id = Column(Integer, primary_key=True, autoincrement=True) username = Column(String(50), unique=True, nullable=False, index=True) email = Column(String(120), unique=True, nullable=False) full_name = Column(String(100)) is_active = Column(Boolean, default=True) score = Column(Float, default=0.0) created_at = Column(DateTime, default=datetime.utcnow) updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow) def __repr__(self): return f"" # Create all tables Base.metadata.create_all(engine)

Relationships

One-to-Many

from sqlalchemy.orm import relationship class Department(Base): __tablename__ = "departments" id = Column(Integer, primary_key=True) name = Column(String(100), nullable=False) employees = relationship("Employee", back_populates="department", lazy="select", cascade="all, delete-orphan") class Employee(Base): __tablename__ = "employees" id = Column(Integer, primary_key=True) name = Column(String(100)) department_id = Column(Integer, ForeignKey("departments.id")) department = relationship("Department", back_populates="employees")

Many-to-Many

from sqlalchemy import Table post_tags = Table( "post_tags", Base.metadata, Column("post_id", Integer, ForeignKey("posts.id"), primary_key=True), Column("tag_id", Integer, ForeignKey("tags.id"), primary_key=True) ) class Post(Base): __tablename__ = "posts" id = Column(Integer, primary_key=True) title = Column(String(200)) tags = relationship("Tag", secondary=post_tags, back_populates="posts") class Tag(Base): __tablename__ = "tags" id = Column(Integer, primary_key=True) name = Column(String(50), unique=True) posts = relationship("Post", secondary=post_tags, back_populates="tags")

CRUD Operations

Create

from sqlalchemy import select from sqlalchemy.orm import Session # Single insert with Session(engine) as session: user = User(username="alice", email="alice@example.com") session.add(user) session.commit() session.refresh(user) print(user.id) # Bulk insert with Session(engine) as session: session.add_all([ User(username="bob", email="bob@example.com"), User(username="charlie", email="charlie@example.com"), ]) session.commit()

Read / Query

from sqlalchemy import select, and_, or_ with Session(engine) as session: # Get by primary key user = session.get(User, 1) # Simple select all users = session.execute(select(User)).scalars().all() # With filter stmt = select(User).where(User.is_active == True) active_users = session.execute(stmt).scalars().all() # Multiple conditions stmt = select(User).where( and_(User.is_active == True, User.score > 50) ) # LIKE / ILIKE stmt = select(User).where(User.email.ilike("%@gmail.com")) # IN stmt = select(User).where(User.id.in_([1, 2, 3])) # Ordering & Pagination stmt = select(User).order_by(User.created_at.desc()).offset(20).limit(10)

Update

from sqlalchemy import update with Session(engine) as session: # ORM style user = session.get(User, 1) if user: user.email = "newemail@example.com" session.commit() # Bulk UPDATE stmt = ( update(User) .where(User.is_active == False) .values(score=0.0) ) session.execute(stmt) session.commit()

Delete

from sqlalchemy import delete with Session(engine) as session: # ORM style user = session.get(User, 1) if user: session.delete(user) session.commit() # Bulk DELETE stmt = delete(User).where(User.is_active == False) session.execute(stmt) session.commit()

Flask Integration

pip install flask flask-sqlalchemy flask-migrate from flask import Flask from flask_sqlalchemy import SQLAlchemy app = Flask(__name__) app.config["SQLALCHEMY_DATABASE_URI"] = "postgresql://user:pass@localhost/mydb" db = SQLAlchemy(app) class User(db.Model): __tablename__ = "users" id = db.Column(db.Integer, primary_key=True) username = db.Column(db.String(50), unique=True, nullable=False) email = db.Column(db.String(120), unique=True, nullable=False) with app.app_context(): db.create_all()

FastAPI Integration

from fastapi import FastAPI, Depends from sqlalchemy.orm import Session from pydantic import BaseModel app = FastAPI() def get_db(): db = SessionLocal() try: yield db finally: db.close() class UserCreate(BaseModel): username: str email: str @app.post("/users/") def create_user(user: UserCreate, db: Session = Depends(get_db)): db_user = User(**user.dict()) db.add(db_user) db.commit() db.refresh(db_user) return db_user

Performance & Optimization

Use selectinload / joinedload to avoid N+1 queries

Add indexes on frequently filtered/sorted columns

Set expire_on_commit=False to avoid re-fetching after commit

Use bulk operations for large inserts

Configure connection pooling appropriately
