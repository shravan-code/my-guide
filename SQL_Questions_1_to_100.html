Q1. Find the second highest salary

📝  Explanation: The inner subquery finds the overall maximum salary. The outer MAX then returns the highest value strictly below that — giving the second highest.

🔷   PostgreSQL Query

SELECT MAX(salary) AS SecondHighestSalary

FROM employees

WHERE salary < (SELECT MAX(salary) FROM employees);

Q9. Longest consecutive login streak per user

📝  Explanation: Subtracting ROW_NUMBER × 1-day interval from the login_date groups consecutive dates under the same grp. COUNT(*) per group gives the streak length.

🔷   PostgreSQL Query

WITH login_dates AS (

  SELECT user_id, login_date,

    login_date - ROW_NUMBER() OVER

      (PARTITION BY user_id ORDER BY login_date)

      * INTERVAL '1 day' AS grp

  FROM user_logins

)

SELECT user_id, COUNT(*) AS streak_length,

       MIN(login_date) AS start_date, MAX(login_date) AS end_date

FROM login_dates

GROUP BY user_id, grp

ORDER BY streak_length DESC;

Q10. Recursive query: full reporting chain for each employee

  SELECT id, name, manager_id, 1 AS level

  FROM employees WHERE manager_id IS NULL

Q31. Detect circular references in the employee-manager hierarchy

📝  Explanation: The ARRAY tracks visited IDs in the path. If an employee's own ID reappears in its ancestry path, a cycle is detected.

🔷   PostgreSQL Query

WITH RECURSIVE mgr_path(id, manager_id, path) AS (

  SELECT id, manager_id, ARRAY[id]

  FROM employees WHERE manager_id IS NOT NULL

  UNION ALL

  SELECT e.id, e.manager_id, path || e.id

  FROM employees e

  JOIN mgr_path mp ON e.manager_id = mp.id

  WHERE NOT e.id = ANY(path)

)

SELECT DISTINCT id FROM mgr_path WHERE id = ANY(path);

Q34. Employees whose salary is a prime number

📝  Explanation: Generates all integers up to max salary, removes composites by finding divisors, leaving only primes.

🔷   PostgreSQL Query

WITH primes AS (

  SELECT generate_series(2, (SELECT MAX(salary) FROM employees)) AS num

  EXCEPT

  SELECT DISTINCT s.num

  FROM generate_series(2, (SELECT MAX(salary) FROM employees)) s(num)

  JOIN generate_series(2, FLOOR(SQRT((SELECT MAX(salary) FROM employees)))::int) d(div)

    ON s.num % d.div = 0 AND s.num <> d.div

)

SELECT * FROM employees WHERE salary IN (SELECT num FROM primes);
