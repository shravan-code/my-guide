Reading & Writing

CSV

Read & Write CSV

Handle CSV files.

import pandas as pd # Read CSV df = pd.read_csv('data.csv') df = pd.read_csv('data.csv', dtype={'id': 'int64', 'name': 'str'}, parse_dates=['created_at'], na_values=['NA', 'N/A', '']) # Write CSV df.to_csv('output.csv', index=False) df.to_csv('output.csv', columns=['id', 'name'], encoding='utf-8')
Excel

Read & Write Excel

Handle Excel files.

# Read Excel df = pd.read_excel('file.xlsx') df = pd.read_excel('file.xlsx', sheet_name='Sheet1', skiprows=2, usecols='A:C') # Multiple sheets dfs = pd.read_excel('file.xlsx', sheet_name=None) # Write Excel df.to_excel('output.xlsx', index=False) df.to_excel('output.xlsx', sheet_name='Data')
JSON

Read & Write JSON

Handle JSON files.

# Read JSON df = pd.read_json('data.json') df = pd.read_json('data.json', orient='records') # JSON lines (one object per line) df = pd.read_json('data.jsonl', lines=True) # Write JSON df.to_json('output.json', orient='records') df.to_json('output.jsonl', orient='records', lines=True)
SQL

Read & Write SQL

Database operations.

from sqlalchemy import create_engine engine = create_engine('postgresql://user:pass@host/db') # Read SQL df = pd.read_sql('SELECT * FROM users', engine) df = pd.read_sql_table('users', engine) df = pd.read_sql_query('SELECT * FROM users WHERE age > 18', engine) # Write SQL df.to_sql('users', engine, if_exists='append', index=False) df.to_sql('users', engine, if_exists='replace', index=False)

Selection & Filtering

Columns

Select Columns

Select and access columns.

# Single column (returns Series) df['name'] df.name # Multiple columns (returns DataFrame) df[['name', 'age', 'city']] # By dtype df.select_dtypes(include=['int64', 'float64']) df.select_dtypes(exclude=['object']) # Columns attribute df.columns.tolist() df.filter(items=['name', 'age']) df.filter(like='date') # Columns containing 'date' df.filter(regex='^id') # Columns starting with 'id'
LOC

LOC & ILOC

Label and position-based selection.

# LOC (label-based) df.loc[0] # Row with index 0 df.loc[0:5] # Rows 0 to 5 (inclusive) df.loc[0:5, 'name'] # Column 'name', rows 0-5 df.loc[0:5, ['name', 'age']] df.loc[df['age'] > 18] # Filter by condition # ILOC (integer position-based) df.iloc[0] # First row df.iloc[0:5] # First 5 rows (exclusive) df.iloc[0:5, 0:3] # First 5 rows, first 3 cols df.iloc[[0, 2, 4]] # Specific rows df.iloc[:, 1] # Second column
Filter

Filter Rows

Filter DataFrames by conditions.

# Single condition df[df['age'] > 18] df.query('age > 18') # Multiple conditions (AND) df[(df['age'] > 18) & (df['city'] == 'NYC')] # Multiple conditions (OR) df[(df['status'] == 'active') | (df['premium'] == True)] # String contains df[df['name'].str.contains('John', case=False)] # Is in list df[df['city'].isin(['NYC', 'LA', 'SF'])] # Between df[df['age'].between(18, 30)] # Not null df[df['name'].notna()] df.dropna(subset=['name', 'email'])
Query

Query Method

Use string expressions for filtering.

# Basic query df.query('age > 18 and city == "NYC"') # Using variables min_age = 21 df.query('age > @min_age') # Multiple conditions df.query('age > 18 and status == ["active", "pending"]') # Date query df.query('created_at > "2024-01-01"') # Inverse (not) df.query('not (age < 18)') # Between shorthand df.query('18 <= age <= 30')

GroupBy & Aggregation

GroupBy

Basic GroupBy

Group and aggregate data.

# Single column groupby df.groupby('category') df.groupby('category')['amount'].sum() df.groupby('category')['amount'].agg('sum') # Multiple columns df.groupby(['category', 'status']) # Aggregation functions df.groupby('category').agg({ 'amount': ['sum', 'mean', 'count'], 'user_id': 'nunique', 'created_at': 'max' }).reset_index() # Size (counts) df.groupby('category').size() df.groupby('category').value_counts()
Agg

Aggregation Methods

Common aggregation functions.

# Single aggregation df['amount'].sum() df['amount'].mean() df['amount'].median() df['amount'].std() df['amount'].var() df['amount'].min() df['amount'].max() df['amount'].count() # Multiple aggregations df.agg({ 'amount': ['sum', 'mean', 'std'], 'age': ['mean', 'min', 'max'] }) # Custom aggregation df.groupby('category')['amount'].agg(lambda x: x.quantile(0.75))
Transform

Transform & Apply

Apply functions to groups.

# Transform (returns same shape) df['category_total'] = df.groupby('category')['amount'].transform('sum') df['category_mean'] = df.groupby('category')['amount'].transform('mean') df['pct_of_group'] = df['amount'] / df.groupby('category')['amount'].transform('sum') # Apply df.groupby('category').apply(lambda x: x.nlargest(3, 'amount')) # Map/Apply on rows df['name_upper'] = df['name'].apply(str.upper) df['full_name'] = df.apply(lambda row: f"{row['first']} {row['last']}", axis=1)
Pivot

Pivot Tables

Reshape data with pivots.

# Pivot table pd.pivot_table(df, values='amount', index='category', columns='status', aggfunc='sum', fill_value=0, margins=True) # Add totals # Cross tabulation pd.crosstab(df['category'], df['status']) pd.crosstab(df['category'], df['status'], normalize='index') # Row % # Melt (wide to long) pd.melt(df, id_vars=['id', 'name'], value_vars=['jan', 'feb', 'mar'], var_name='month', value_name='revenue')

Merge & Join

Merge

Merge DataFrames

Combine DataFrames.

# Inner join (default) pd.merge(df1, df2, on='id', how='inner') # Left join pd.merge(df1, df2, on='id', how='left') # Right join pd.merge(df1, df2, on='id', how='right') # Outer join pd.merge(df1, df2, on='id', how='outer') # Different column names pd.merge(df1, df2, left_on='user_id', right_on='id', how='left') # Multiple keys pd.merge(df1, df2, on=['id', 'date'], how='inner') # Suffix for overlapping column names pd.merge(df1, df2, on='id', suffixes=('_left', '_right'))
Join

Join on Index

Join using index.

# Join on index df1.join(df2) # Join on specific column df1.join(df2, lsuffix='_l', rsuffix='_r') # Concat (stack DataFrames) pd.concat([df1, df2]) # Vertical stack pd.concat([df1, df2], axis=1) # Horizontal stack pd.concat([df1, df2], ignore_index=True) # Reset index # Append (deprecated, use concat) # df1.append(df2) # Use pd.concat instead
Merge AsOf

Merge AsOf

Time-based or ordered merges.

# Merge asof (nearest match) pd.merge_asof( trades, # Left DataFrame quotes, # Right DataFrame on='timestamp', # Merge key by='symbol', # Group by direction='nearest', # 'forward', 'backward', 'nearest' tolerance=pd.Timedelta('1min') ) # Example use case: match trades to nearest quote # trades: timestamps of trades # quotes: timestamps of quotes # Result: each trade matched to nearest quote
Concat

Concatenate DataFrames

Stack multiple DataFrames.

# Vertical concat pd.concat([df1, df2, df3], ignore_index=True) # Horizontal concat pd.concat([df1, df2], axis=1) # With keys (hierarchical index) pd.concat([df1, df2], keys=['source1', 'source2']) # Only common columns pd.concat([df1, df2], join='inner') # Dictionary of DataFrames pd.concat({'file1': df1, 'file2': df2}) # Along specific axis with keys result = pd.concat([df1, df2], axis=0, keys=['Q1', 'Q2'], names=['quarter'])

Data Cleaning

Missing

Handle Missing Values

Detect and handle nulls.

# Detect df.isnull().sum() df.isna().sum() df.info() # Shows non-null counts # Drop df.dropna() # Drop rows with any null df.dropna(subset=['name']) # Drop if name is null df.dropna(how='all') # Drop only if all null df.dropna(thresh=2) # Keep rows with at least 2 non-null # Fill df.fillna(0) # Fill with 0 df.fillna({'age': 0, 'city': 'Unknown'}) # Different fills df.fillna(method='ffill') # Forward fill df.fillna(method='bfill') # Backward fill df.fillna(df.mean()) # Fill with column means # Interpolate df.interpolate() # Linear interpolation
Duplicates

Handle Duplicates

Find and remove duplicates.

# Detect duplicates df.duplicated().sum() df.duplicated(subset=['email']).sum() df.duplicated(keep='last') # Mark all but last as dup # Remove duplicates df.drop_duplicates() df.drop_duplicates(subset=['email']) df.drop_duplicates(subset=['name', 'email'], keep='last') # Get duplicates dups = df[df.duplicated(subset=['email'], keep=False)] # Count by column df['email'].value_counts() df['email'].value_counts().head(10)
Types

Data Type Conversion

Convert column data types.

# Check types df.dtypes df.info() # Convert types df['age'] = df['age'].astype(int) df['price'] = df['price'].astype(float) df['is_active'] = df['is_active'].astype(bool) # To datetime df['date'] = pd.to_datetime(df['date']) df['date'] = pd.to_datetime(df['date'], format='%Y-%m-%d') df['date'] = pd.to_datetime(df['date'], errors='coerce') # Invalid -> NaT # Extract datetime components df['year'] = df['date'].dt.year df['month'] = df['date'].dt.month df['day_of_week'] = df['date'].dt.dayofweek
Strings

String Operations

Text manipulation with .str accessor.

# Case conversion df['name'] = df['name'].str.lower() df['name'] = df['name'].str.upper() df['name'] = df['name'].str.title() df['name'] = df['name'].str.strip() # Search and replace df['name'].str.contains('John') df['name'].str.startswith('Mr') df['name'].str.replace('Mr.', '') df['name'].str.replace(r'\d+', '', regex=True) # Split and extract df['name'].str.split() df['email'].str.split('@').str[1] # Domain df['address'].str.extract(r'(\d{5})') # Extract zip # Length and slice df['name'].str.len() df['name'].str[:10]

Datetime Operations

Parsing

Datetime Parsing

Convert strings to datetime.

# Parse dates df['date'] = pd.to_datetime(df['date']) df['date'] = pd.to_datetime(df['date'], format='%Y-%m-%d') # Multiple formats df['date'] = pd.to_datetime(df['date'], format='mixed', dayfirst=True) # Create date range dates = pd.date_range('2024-01-01', periods=365, freq='D') dates = pd.date_range('2024-01-01', '2024-12-31', freq='M') # Extract components df['year'] = df['date'].dt.year df['month'] = df['date'].dt.month df['day'] = df['date'].dt.day df['dayofweek'] = df['date'].dt.dayofweek # Monday=0 df['quarter'] = df['date'].dt.quarter
Arithmetic

Datetime Arithmetic

Perform date calculations.

# Time differences df['days_since'] = (pd.Timestamp.now() - df['date']).dt.days df['diff'] = df['end_date'] - df['start_date'] # Add/subtract df['next_week'] = df['date'] + pd.Timedelta(days=7) df['prev_month'] = df['date'] - pd.DateOffset(months=1) # Date offset from pandas.tseries.offsets import BusinessDay, MonthEnd df['next_biz'] = df['date'] + BusinessDay() df['month_end'] = df['date'] + MonthEnd() # Floor/ceil/round df['date'].dt.floor('D') # Floor to day df['date'].dt.ceil('H') # Ceil to hour df['date'].dt.round('min') # Round to minute
Resample

Time Series Resampling

Aggregate time series data.

# Set date as index for resampling df = df.set_index('date') # Resample and aggregate df.resample('D').sum() # Daily sum df.resample('W').mean() # Weekly mean df.resample('M').agg({ 'amount': 'sum', 'quantity': 'mean' }) # Resample options # 'D' - daily, 'W' - weekly, 'M' - month end # 'ME' - month end (new), 'MS' - month start # 'Q' - quarter, 'Y' - year, 'h' - hour # 'min' - minute, 's' - second # Rolling window df['rolling_7d'] = df['amount'].rolling('7D').sum() df['rolling_avg'] = df['amount'].rolling(window=7).mean()
Period

Period & Interval

Work with time periods.

# Create periods periods = pd.period_range('2024-01', periods=12, freq='M') df['period'] = pd.PeriodIndex(df['date'], freq='M') # Convert period to timestamp df['period'].dt.to_timestamp() # Date intervals interval = pd.Interval(left='2024-01-01', right='2024-12-31') df['in_period'] = df['date'].apply( lambda x: x in interval if pd.notna(x) else False )
