PySpark Cheatsheet

Quick reference for Apache Spark with PySpark DataFrame API and common operations.

Session & Configuration

Session

Create Spark Session

Initialize Spark with configuration options.

from pyspark.sql import SparkSession from pyspark.sql import functions as F spark = SparkSession.builder \\ .appName("MyApp") \\ .master("local[*]") \\ .config("spark.executor.memory", "4g") \\ .config("spark.driver.memory", "2g") \\ .config("spark.sql.shuffle.partitions", "200") \\ .config("spark.sql.adaptive.enabled", "true") \\ .getOrCreate() # Stop session spark.stop()
Config

Common Configurations

Useful Spark configurations.

# Memory spark.conf.set("spark.executor.memory", "8g") spark.conf.set("spark.driver.memory", "4g") # Shuffle spark.conf.set("spark.sql.shuffle.partitions", "200") # Adaptive query execution spark.conf.set("spark.sql.adaptive.enabled", "true") # View all configs spark.conf.getAll()

Reading Data

Parquet

Read Parquet

Read Parquet files with options.

# Basic read df = spark.read.parquet("s3://bucket/data/") # Multiple paths df = spark.read.parquet("path1/", "path2/") # With schema merge df = spark.read \\ .option("mergeSchema", "true") \\ .parquet("s3://bucket/data/") # Specific columns df = spark.read.parquet("path/") \\ .select("col1", "col2")

Read CSV

Read CSV files with options.

df = spark.read \\ .option("header", "true") \\ .option("inferSchema", "true") \\ .option("delimiter", ",") \\ .option("nullValue", "NULL") \\ .csv("path/to/file.csv") # With explicit schema from pyspark.sql.types import * schema = StructType([ StructField("id", IntegerType()), StructField("name", StringType()) ]) df = spark.read.schema(schema).csv("path/")

Read from Database

Read from relational databases.

df = spark.read \\ .format("jdbc") \\ .option("url", "jdbc:postgresql://host:5432/db") \\ .option("dbtable", "schema.table") \\ .option("user", "user") \\ .option("password", "pass") \\ .option("driver", "org.postgresql.Driver") \\ .load() # With partitioning df = spark.read \\ .format("jdbc") \\ .option("url", url) \\ .option("dbtable", table) \\ .option("partitionColumn", "id") \\ .option("lowerBound", 1) \\ .option("upperBound", 1000000) \\ .option("numPartitions", 10) \\ .load()

Read JSON files.

Transformations

Select

Select & WithColumn

Select columns and add new ones.

# Select columns df.select("name", "age") df.select(df.name, df.age) df.select(F.col("name"), F.col("age")) # Select with alias df.select(F.col("name").alias("full_name")) # Add column df.withColumn("new_col", F.lit("default")) df.withColumn(" doubled", F.col("amount") * 2) # Conditional column df.withColumn("status", F.when(F.col("age") > 18, "adult") .otherwise("minor"))
Filter

Filter & Where

Filter rows based on conditions.

# Basic filter df.filter(F.col("age") > 18) df.where("age > 18") # Multiple conditions df.filter((F.col("age") > 18) & (F.col("city") == "NYC")) # OR condition df.filter((F.col("status") == "active") | (F.col("premium") == True)) # Null check df.filter(F.col("email").isNotNull()) df.filter(F.col("name").isNull()) # In list df.filter(F.col("city").isin("NYC", "LA", "SF"))
GroupBy

Aggregation

Group and aggregate data.

# Basic aggregation df.groupBy("category").count() df.groupBy("category").sum("amount") df.groupBy("category").agg( F.sum("amount").alias("total"), F.avg("amount").alias("avg"), F.count("*").alias("count"), F.max("amount").alias("max"), F.min("amount").alias("min") ) # Multiple group by df.groupBy("category", "status") \\ .agg(F.sum("amount").alias("total"))
Distinct

Distinct & DropDuplicates

Remove duplicate rows.

# All columns distinct df.distinct() # Drop duplicates on specific columns df.dropDuplicates(["id"]) df.dropDuplicates(["name", "email"]) # Dropna (remove nulls) df.na.drop() df.na.drop(subset=["name", "email"]) # Fill nulls df.na.fill(0) df.na.fill({"age": 0, "city": "Unknown"})

Joins

Join Types

Join DataFrames

Join DataFrames with different types.

# Inner join df1.join(df2, "id", "inner") # Left join df1.join(df2, "id", "left") df1.join(df2, df1.id == df2.user_id, "left") # Right join df1.join(df2, "id", "right") # Full outer join df1.join(df2, "id", "outer") # Cross join df1.crossJoin(df2)
Conditions

Complex Join Conditions

Joins with multiple conditions.

# Multiple conditions df1.join(df2, (df1.id == df2.user_id) & (df1.date == df2.created_date), "inner") # Rename after join df1.join(df2, df1.id == df2.user_id, "left") \\ .select(df1.id, df1.name, df2.amount) # Anti join (rows in df1 not in df2) df1.join(df2, "id", "anti") # Semi join (rows in df1 that match df2) df1.join(df2, "id", "semi")
Broadcast

Broadcast Join

Optimize joins with small tables.

from pyspark.sql.functions import broadcast # Broadcast small DataFrame result = large_df.join(broadcast(small_df), "id") # SQL hint df.createOrReplaceTempView("large") small_df.createOrReplaceTempView("small") spark.sql(""" SELECT /*+ BROADCAST(small) */ * FROM large JOIN small ON large.id = small.id """)

Writing Data

Parquet

Write Parquet

Write DataFrames to Parquet format.

# Basic write df.write.parquet("s3://bucket/output/") # Overwrite mode df.write.mode("overwrite").parquet("path/") # Append mode df.write.mode("append").parquet("path/") # Partitioned write df.write \\ .mode("overwrite") \\ .partitionBy("date", "category") \\ .parquet("s3://bucket/output/") # Coalesce (single file) df.coalesce(1).write.parquet("path/")
JDBC

Write to Database

Write to relational databases.

df.write \\ .format("jdbc") \\ .option("url", "jdbc:postgresql://host/db") \\ .option("dbtable", "schema.table") \\ .option("user", "user") \\ .option("password", "pass") \\ .mode("overwrite") \\ .save() # Append mode df.write \\ .format("jdbc") \\ .option("url", url) \\ .option("dbtable", table) \\ .mode("append") \\ .save()
Modes

Write Modes

Different write behaviors.

# overwrite - Replace existing df.write.mode("overwrite").save(path) # append - Add to existing df.write.mode("append").save(path) # error - Fail if exists (default) df.write.mode("error").save(path) # ignore - Skip if exists df.write.mode("ignore").save(path)
Bucket

Bucketed Tables

Write bucketed tables for optimization.

df.write \\ .bucketBy(16, "user_id") \\ .sortBy("created_at") \\ .mode("overwrite") \\ .saveAsTable("my_bucketed_table") # Read bucketed table df = spark.table("my_bucketed_table")

UDFs & Functions

UDF

User Defined Functions

Create custom functions for Spark.

from pyspark.sql.functions import udf from pyspark.sql.types import StringType, IntegerType # Define UDF @udf(StringType()) def extract_domain(email): if email and "@" in email: return email.split("@")[1] return None # Register UDF spark.udf.register("extract_domain", lambda x: x.split("@")[1] if x else None, StringType()) # Use UDF df.withColumn("domain", extract_domain(F.col("email")))
Pandas UDF

Pandas UDF (Vectorized)

Fast UDFs using pandas/vectorized operations.

from pyspark.sql.functions import pandas_udf from pyspark.sql.types import DoubleType import pandas as pd @pandas_udf(DoubleType()) def pandas_mean(series: pd.Series) -> float: return series.mean() # Apply df.groupBy("category") \\ .agg(pandas_mean(F.col("amount"))) # Series to Series @pandas_udf(StringType()) def uppercase_udf(s: pd.Series) -> pd.Series: return s.str.upper()
Window

Window Functions

Analytics with window specifications.

from pyspark.sql.window import Window window = Window.partitionBy("category") \\ .orderBy(F.col("amount").desc()) # Rank and row number df.withColumn("rank", F.rank().over(window)) df.withColumn("row_num", F.row_number().over(window)) df.withColumn("dense_rank", F.dense_rank().over(window)) # Lag and lead df.withColumn("prev_amount", F.lag("amount", 1).over(window)) df.withColumn("next_amount", F.lead("amount", 1).over(window)) # Running total window_unbounded = Window.partitionBy("category") \\ .orderBy("date") \\ .rowsBetween(Window.unboundedPreceding, 0) df.withColumn("running_total", F.sum("amount").over(window_unbounded))

Built-in Functions

Common built-in functions.

Performance

Repartition

Repartition & Coalesce

Control partitioning for performance.

# Repartition (shuffle) df.repartition(200) df.repartition("category") # by column df.repartition(200, "category") # num + column # Coalesce (no shuffle) df.coalesce(10) # reduce partitions # Check partitions df.rdd.getNumPartitions() # Cache df.cache() # or df.persist()
Caching

Cache & Persist

Cache DataFrames for reuse.

from pyspark import StorageLevel # Cache (in memory) df.cache() # Persist with level df.persist(StorageLevel.MEMORY_AND_DISK) df.persist(StorageLevel.DISK_ONLY) # Unpersist df.unpersist() # Check cached spark.catalog.isCached("my_table") # Cache table df.createOrReplaceTempView("my_table") spark.catalog.cacheTable("my_table")

Analyze query execution plans.

# Show physical plan df.explain() # Show all plans df.explain("extended") # parsed, analyzed, optimized, physical df.explain("simple") # physical only # Print schema df.printSchema() # Show stats df.describe().show() # Count rows df.count()
AQE

Adaptive Query Execution

Enable AQE for automatic optimization.

# Enable AQE spark.conf.set("spark.sql.adaptive.enabled", "true") # Coalesce partitions spark.conf.set("spark.sql.adaptive.coalescePartitions.enabled", "true") spark.conf.set("spark.sql.adaptive.coalescePartitions.minPartitionSize", "1MB") # Skew join optimization spark.conf.set("spark.sql.adaptive.skewJoin.enabled", "true") spark.conf.set("spark.sql.adaptive.skewJoin.skewedPartitionFactor", "5") # Check AQE in explain df.explain() # Look for "AdaptiveSparkPlan"
